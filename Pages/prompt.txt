### Componentes DevExpress Estándar:
```razor
<!-- Inputs básicos -->
<DxTextBox @bind-Text="..." NullText="..." />
<DxComboBox Data="..." @bind-Value="..." TextFieldName="Text" ValueFieldName="Value" />
<DxCheckBox @bind-Checked="..." Text="..." />
<DxDateEdit @bind-Date="..." DisplayFormat="dd/MM/yyyy" />
<DxSpinEdit @bind-Value="..." MinValue="0" DisplayFormat="N2" />

<!-- Búsqueda con botón -->
<div class="search-field">
    <DxTextBox @bind-Text="...">
        <Buttons>
            <DxEditorButton IconCssClass="fi fi-rr-search" Click="@OnSearch..." />
        </Buttons>
    </DxTextBox>
</div>

<!-- Tabs cuando aparecen en imagen -->
<DxTabs @bind-ActiveTabIndex="...">
    <DxTabPage Text="...">...</DxTabPage>
</DxTabs>

<!-- Grids para datos tabulares -->
<DxGrid Data="..." CssClass="...-data-grid">
    <Columns>
        <DxGridDataColumn FieldName="..." Caption="..." />
    </Columns>
</DxGrid>
```
